<script>
	var btn=document.getElementsByTagName('button');
	var input=document.getElementsByTagName('input')[0];
	for(var i = 0; i < 11; i++) {
		btn[i].onclick=function (argument) {
			input.value=this.innerHTML
		}
	}
</script>	

<script>
	var btn_1=document.getElementById('box1').getElementsByTagName('button');
	var input_1=document.getElementsByTagName('input')[1];
	for(var i = 1; i < 10; i++) {
		btn_1[i].onclick=function (argument) {
			if(input_1.value=='0'){
				input_1.value='';
			}
			input_1.value+=this.innerHTML
		}
	}
	btn_1[0].onclick=function (argument) {
		if(input_1.value=='0'){
			input_1.value+=''
		}else{
			input_1.value+='0'
		}
	}
	btn_1[10].onclick=function (argument) {
		if(input_1.value==''){
			input_1.value+='0.'
		}else{
			input_1.value+='.'
		}
	}
</script>
