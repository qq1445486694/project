function getutag(utag) {
		var utag=document.getElementsByTagName(utag);
		return utag;
	}
var lt=getutag('div');
let sixarray=[19,17,82,52,1080];

数组

lt[0].innerHTML+=sixarray;

push尾部添加： 

sixarray.push('1');
lt[1].innerHTML+=sixarray;

unshift向前添加

sixarray.unshift('2060');
lt[2].innerHTML+=sixarray;

pop尾部删除

sixarray.pop();
lt[3].innerHTML+=sixarray;

shift头部删除

sixarray.shift();
lt[4].innerHTML+=sixarray;

sort首字母正排序

var sixarray_2=['alice','ordiric','31','obnixilis','9821','charich','zzward','alexchander']
sixarray_2.sort();
lt[5].innerHTML+=sixarray_2;

数字从小到大排序

function sortnum(a,b) {
		return a-b
	}
lt[6].innerHTML+=sixarray.sort(sortnum);

concat链接数组

lt[7].innerHTML+=sixarray.concat(sixarray_2);

join插入

lt[8].innerHTML+=sixarray.join('virus');

splice两个参数时(从第几个之后开始，显示多少个)

lt[9].innerHTML+=sixarray.splice(2,3);

splice多个参数时(从第几个之后开始，删除多少个，添加数据)

var sixarray_3=[8,54,298,15,2,571,79];
sixarray_3.splice(3,2,4,4,4,4);
lt[10].innerHTML+=sixarray_3;

reverse()数组反转

var sixarray_4=[100,1060,2060,980,1660,1070];
lt[11].innerHTML+=sixarray_4+'<br/>';
