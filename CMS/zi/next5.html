	function getEvent(){
    if(window.event)    {return window.event;}
    func=getEvent.caller;
    while(func!=null){
        var arg0=func.arguments[0];
        if(arg0){
            if((arg0.constructor==Event || arg0.constructor ==MouseEvent
                || arg0.constructor==KeyboardEvent)
                ||(typeof(arg0)=="object" && arg0.preventDefault
                && arg0.stopPropagation)){
                return arg0;
            }
        }
        func=func.caller;
    }
    return null;
}
//阻止冒泡
function cancelBubble()
{
    var e=getEvent();
    if(window.event){
        //e.returnValue=false;//阻止自身行为
        e.cancelBubble=true;//阻止冒泡
    }else if(e.preventDefault){
        //e.preventDefault();//阻止自身行为
        e.stopPropagation();//阻止冒泡
    }
}

继承

<script>
	function shujua(name,sex) {
		this.name=name;
		this.sex=sex;
	}
	shujua.prototype.showname=function (argument) {
		alert(this.name)
	}
	shujua.prototype.showsex=function (argument) {
		alert(this.sex)
	}
	var fan=new shujua('樊钦志','男');
	// fan.showsex();
	function shujuan (name,sex,job) {
		shujua.call(this,name,sex);
		this.job=job;
	}
	shujuan.prototype=shujua.prototype;
	shujuan.prototype.showjob=function (argument) {
		alert(this.job);
	};
	var ojob1=new shujuan('张俊涛','男','大学生');

</script>

继承方法2(因为某行文字忘注释还找了半天的那个)

<script>
	function hittt (zidingyi) {
		var _this=this;
		this.oElement=document.getElementById(zidingyi);
		this.oElement.onmouseover=function (argument) {
			_this.changecolor();
		}
	}
	hittt.prototype.changecolor=function  (argument) {
		this.oElement.style.background='skyblue';
	}
	// new hittt('minibox_div');
	// 一个基本的函数,然后我们要创建一个基于这个函数的函数
	function hitan (zidingyi) {
		hittt.call(this,zidingyi);
	}
	hitan.prototype=hittt.prototype;
	new hitan('minibox_div');
</script>
