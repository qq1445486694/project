<script>
	var ul3=document.getElementById('ul3');
	var oli=ul3.getElementsByTagName('li');
	ul3.innerHTML+=ul3.innerHTML; 无缝切换起码有两套完全相同的列表
	ul3.style.width=oli[0].offsetWidth*oli.length+5+'px'; 因为火狐诡异的兼容问题需要多5像素 UL的宽度为所有项的宽度和
	setInterval(function (argument) {
		ul3.style.left=ul3.offsetLeft-3+'px'; 
		if(ul3.offsetLeft<-ul3.offsetWidth/2){ 判断是否为切换的那个定位值，是就重置位置
			ul3.style.left=0
		}
	},30)
	</script>
	
	<script>
	var let=document.getElementById('let');
	var rit=document.getElementById('rit');
	var ul4=document.getElementById('ul4');
	var ali=ul4.getElementsByTagName('li');
	var ipseed=-3;
	ul4.innerHTML+=ul4.innerHTML;
	let.onclick=function (argument) {
		ipseed=3;
	}
	rit.onclick=function (argument) {
		ipseed=-3
	}
	bigbox5.onmouseout=function (argument) {
		timer1
	}
	bigbox5.onmouseover=function (argument) {
		clearInterval(timer1)
	}
	ul4.style.width=ali[0].offsetWidth*ali.length+5+'px';
	timer1=setInterval(function (argument) {
		ul4.style.left=ul4.offsetLeft+ipseed+'px';
		if(ul4.offsetLeft<-ul4.offsetWidth/2){
			ul4.style.left='0px';
		}else if(ul4.offsetLeft>0){
			ul4.style.left=-ul4.offsetWidth/2+'px';
		}
	},30)
	</script>
